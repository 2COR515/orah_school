// 7. shouldSendReminder logic
function shouldSendReminder(frequency) {
  const day = new Date().getDay(); // 0-6
  
  if (frequency === 'daily') return true;
  if (frequency === 'weekly') return day === 1;
  if (frequency === 'twice-weekly') return day === 1 || day === 4;
  
  return false; // Default: don't send
}
